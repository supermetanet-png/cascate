
user  nginx;
worker_processes  auto;

error_log  /var/log/nginx/error.log warn;
pid        /var/run/nginx.pid;

events {
    worker_connections  1024;
}

http {
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;

    sendfile        on;
    keepalive_timeout  65;

    # Otimização para Uploads grandes (Importação de dados)
    client_max_body_size 100M;

    server {
        listen 80;
        listen [::]:80;
        server_name _; # Aceita IP ou qualquer domínio

        # Essencial para renovação de SSL No-Code
        location /.well-known/acme-challenge/ {
            root /var/www/certbot;
        }

        # Frontend adaptativo
        location / {
            proxy_pass http://frontend:80;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
            proxy_read_timeout 300;
        }

        # Control Plane (Auth e Projetos)
        location /api/control/ {
            proxy_pass http://backend_control:3000/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # Data Plane (Query e Multi-tenancy)
        location /api/data/ {
            proxy_pass http://backend_data:3000/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }
    }

    # Bloco SSL sugerido (descomentar somente após gerar os .pem via Certbot ou Painel)

    #         ...
