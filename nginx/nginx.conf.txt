
user  nginx;
worker_processes  auto;

error_log  /var/log/nginx/error.log warn;
pid        /var/run/nginx.pid;

events {
    worker_connections  1024;
}

http {
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;

    sendfile        on;
    keepalive_timeout  65;

    # Configuração resisiente: O Nginx ouvirá na 80 sempre.
    # O redirecionamento para 443 deve ser feito apenas quando o certificado existir no disco.
    
    server {
        listen 80;
        listen [::]:80;
        server_name _;

        # Suporte para Let's Encrypt ACME Challenge (Essencial para No-Coders)
        # O Certbot colocará os arquivos em /var/www/certbot
        location /.well-known/acme-challenge/ {
            root /var/www/certbot;
        }

        # Frontend
        location / {
            proxy_pass http://frontend:80;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # Control Plane API
        location /api/control/ {
            proxy_pass http://backend_control:3000/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # Data Plane API
        location /api/data/ {
            proxy_pass http://backend_data:3000/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }
    }

    # BLOCO SSL (Comentar para evitar crash do container Nginx se os arquivos .pem não existirem)
    # server {
    #     listen 443 ssl;
    #     server_name cascata.unibloom.shop;
    #     ssl_certificate /etc/letsencrypt/live/cascata.unibloom.shop/fullchain.pem;
    #     ssl_certificate_key /etc/letsencrypt/live/cascata.unibloom.shop/privkey.pem;
    #     ...
    # }
}
