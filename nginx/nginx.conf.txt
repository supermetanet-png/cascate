
user  nginx;
worker_processes  auto;

error_log  /var/log/nginx/error.log warn;
pid        /var/run/nginx.pid;

events {
    worker_connections  1024;
}

http {
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;

    sendfile        on;
    keepalive_timeout  65;

    # Mapa para verificar se os arquivos SSL existem (Prevenção de Crash)
    # Em uma VPS real, se o arquivo não existe, o Nginx não sobe se o listen 443 estiver ativo.
    # Por isso, mantemos o 80 sempre ativo.

    server {
        listen 80;
        listen [::]:80;
        server_name _;

        # Suporte para Let's Encrypt ACME Challenge (Essencial para No-Coders)
        location /.well-known/acme-challenge/ {
            root /var/www/certbot;
        }

        # Frontend
        location / {
            proxy_pass http://frontend:80;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # Control Plane
        location /api/control/ {
            proxy_pass http://backend_control:3000/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # Data Plane (Corrigido para repassar o path correto)
        location /api/data/ {
            proxy_pass http://backend_data:3000/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }
    }

    # Bloco SSL - Recomendado descomentar apenas após o Certbot confirmar a geração dos arquivos
    # server {

    #     
    #     location / {
    #         proxy_pass http://frontend:80;
    #         ... (mesmas configs do bloco 80)
    #     }
    # }
}
