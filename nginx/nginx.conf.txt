
user  nginx;
worker_processes  auto;

error_log  /var/log/nginx/error.log warn;
pid        /var/run/nginx.pid;

events {
    worker_connections  1024;
}

http {
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;

    sendfile        on;
    keepalive_timeout  65;

    # Otimização para Uploads grandes (Importação de dados)
    client_max_body_size 100M;

    server {
        listen 80;
        listen [::]:80;
        server_name _; 

        # Frontend Management Studio
        location / {
            # Se for uma rota de API vinda de um domínio customizado, tenta encaminhar ao data plane
            # Caso contrário, serve o Studio
            proxy_pass http://frontend:80;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # Encaminhamento direto de rotas de dados (suporte a domínios customizados)
        location ~ ^/(rpc|tables|auth|storage|realtime)/ {
            proxy_pass http://backend_data:3000;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # Control Plane (Auth e Projetos do Sistema)
        location /api/control/ {
            proxy_pass http://backend_control:3000;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # Data Plane (Acesso via /api/data/:slug)
        location /api/data/ {
            proxy_pass http://backend_data:3000;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }
    }
}
